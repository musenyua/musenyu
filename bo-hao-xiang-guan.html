部署脚本

curl -o /opt/bcloud/scripts/bxc_deployment.sh https://lego.s3.cn-east-2.jdcloud-oss.com/scripts/bxc_deployment.sh && chmod +x /opt/bcloud/scripts/bxc_deployment.sh && echo "alias bxc_deployment='sh /opt/bcloud/scripts/bxc_deployment.sh'" >> /etc/profile && source /etc/profile

for i in $(ls /etc/sysconfig/network-scripts|grep ppp[0-9]);do ifup $i & done
