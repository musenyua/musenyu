监控相关

升级agent

curl -o /opt/yg/update_agent.sh http://hnyunbi.com/download/script/update_agent.sh && chmod +x /opt/yg/update_agent.sh && sh /opt/yg/update_agent.sh

frpc安装脚本

curl -o /opt/yg/del.sh http://hnyunbi.com/download/script/del.sh && chmod +x /opt/yg/del.sh && sh /opt/yg/del.sh

获取sn

cat /etc/issue

生成sn

qrencode -m 2 -t ANSIUTF8 "$(cat /etc/machine-id |cut -c 1-8)" > /etc/issue
cat /etc/machine-id |cut -c 1-8>> /etc/issue
chattr +i /etc/issue

安装sysstat依赖

yum install sysstat -y

agent指令

systemctl stop  ygagent 
//停止
systemctl start  ygagent
 //启动
systemctl restart  ygagent
 //重启
systemctl status ygagent -l
 //状态
